import { PartialType } from '@nestjs/swagger';
import { CreateUserDto } from './create-user.dto';
import { UserRole } from 'src/enums/users/user-role.enum';
import { UserGender } from 'src/enums/users/user-gender.enum';
import { IsOptional } from 'class-validator';
import { Weekday } from 'src/enums/shared/weekday.enum';

export class UpdateUserDto extends PartialType(CreateUserDto) {
  @IsOptional()
  username?: string;

  @IsOptional()
  firstName?: string;

  @IsOptional()
  lastName?: string;

  @IsOptional()
  email?: string;

  @IsOptional()
  phoneNumber?: string;

  @IsOptional()
  gender?: UserGender;

  @IsOptional()
  role?: UserRole;

  @IsOptional()
  avatar?: Express.Multer.File;
}
